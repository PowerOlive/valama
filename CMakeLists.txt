cmake_minimum_required(VERSION "2.8.4")

set(TARGET "" CACHE STRING "Target")
set(target_specified FALSE)
if ("${TARGET}" STREQUAL "cmaketest")
  include("CMake_cmaketest.txt")
  set(target_specified TRUE)
endif ()
if ("${TARGET}" STREQUAL "valama-codecontextd")
  include("CMake_valama-codecontextd.txt")
  set(target_specified TRUE)
endif ()
if (NOT target_specified)
  message( FATAL_ERROR "Specify a target via 'cmake -DTARGET=my_target ...'" )
endif ()
